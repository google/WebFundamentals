import {addRoute} from './addRoute.mjs';
import {precache} from './precache.mjs';
import './_version.mjs';


/**
 * This method will add entries to the precache list and add a route to
 * respond to fetch events.
 *
 * This is a convenience method that will call
 * [precache()]{@link module:workbox-precaching.precache} and
 * [addRoute()]{@link module:workbox-precaching.addRoute} in a single call.
 *
 * @param {Array&lt;Object|string>} entries Array of entries to precache.
 * @param {Object} options See
 * [addRoute() options]{@link module:workbox-precaching.addRoute}.
 *
 * @alias workbox.precaching.precacheAndRoute
 */
export const precacheAndRoute = (entries, options) => {
  precache(entries);
  addRoute(options);
};
